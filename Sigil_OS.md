<script>
(function() {
  'use strict';
  
  // Get all activate buttons
  const activateButtons = document.querySelectorAll('.activate-btn');
  
  activateButtons.forEach(button => {
    button.addEventListener('click', function(e) {
      e.stopPropagation();
      
      const card = this.closest('.sigil-card');
      const details = card.querySelector('.sigil-details');
      const isActive = details.classList.contains('active');
      
      // Toggle active state
      if (isActive) {
        details.classList.remove('active');
        card.classList.remove('active');
        this.textContent = 'ACTIVATE SIGIL';
      } else {
        details.classList.add('active');
        card.classList.add('active');
        this.textContent = 'DEACTIVATE SIGIL';
        
        // Smooth scroll to show details
        setTimeout(() => {
          details.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }, 100);
      }
    });
  });
  
  // Optional: Click anywhere on card to toggle
  document.querySelectorAll('.sigil-card').forEach(card => {
    card.addEventListener('click', function(e) {
      // Don't trigger if clicking the button
      if (e.target.classList.contains('activate-btn')) return;
      
      const button = this.querySelector('.activate-btn');
      button.click();
    });
  });
  
})();
</script>
